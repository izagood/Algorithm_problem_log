###   정답(Solution)
```java
import java.util.*;

class Solution {
    public String solution(int[] numbers) {
        
        String[] arr = Arrays.stream(numbers)
            .boxed()
            .map(String::valueOf)
            .sorted((a, b) -> (b+a).compareTo(a+b))
            .toArray(String[]::new);
        
        if(arr[0].equals("0")){
            return "0";
        }
        
        StringBuilder sb = new StringBuilder();
        
        for(String str : arr){
            sb.append(str);
        }
        
        return sb.toString();
    }
}
```
