```java
class Solution {
    public int[] solution(int N, int[] stages) {
        
        /*
        실패율 = 스테이지에 도달했으나 아직 클리어하지 못한 플레이어의 수 / 스테이지에 도달한 플레이어 수
        
        전체 스테이지 수 N
        멈춰 있는 stages
        
        총 플레이어수 stages.length
        실패율 높은 스테이지부터 내림차순 return
        */
        int allPlayer = stages.length;
        
        //멈춰 있는 스테이지 array
        int[] statusArr = new int[N];
        //N+1 모두 깬 플레이어가 있네
        
        for(int i=0; i<N; i++){
            int stageNum = stages[i];
            statusArr[stageNum - 1]++;
        }
        
        int[] failArr = new int[N];
        int failRate = 0;
        
        //실패율 계산
        for(int i=0; i<N ; i++){
            //현재 클리어 못한 수
            //하위는 모두 클리어 한거지
            failRate = statusArr[i] / allPlayer;
            allPlayer -= statusArr[i];
            failArr[i] = failRate;
        }
        
        //정렬에 조건이 있네
        for(int fail: failArr){
            
        System.out.println(fail);
        }
        
        
        
        return null;
    }
}
```
